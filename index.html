function getWeekKey(date = new Date()) { // 获取一年中的第几周（简单版） const start = new Date(date.getFullYear(), 0, 1); const diff = date - start; const oneWeek = 1000 * 60 * 60 * 24 * 7; return Math.floor(diff / oneWeek); } function checkWeeklyPermission(user) { const currentWeek = getWeekKey(); // 用户上次激活的周 const lastWeek = user.lastWeek; // 是否跨周 if (currentWeek !== lastWeek) { user.active = false; // 自动失效 } if (!user.active) { return { access: false, message: "本周权限已过期，请重新激活" }; } return { access: true, message: "权限正常，本周可使用" }; } // 测试 const user = { active: true, lastWeek: 2000 }; console.log(checkWeeklyPermission(user));
